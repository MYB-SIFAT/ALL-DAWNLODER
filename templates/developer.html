POST /api/analyze Analyze a video URL — returns title, thumbnail, platform & duration 
curl -X POST {{ request.host_url.rstrip('/') }}/api/analyze \
  -H "Content-Type: application/json" \
  -d '{"url": "https://www.youtube.com/watch?v=dQw4w9WgXcQ"}'
import requests

res = requests.post("{{ request.host_url.rstrip('/') }}/api/analyze", json={
    "url": "https://www.youtube.com/watch?v=dQw4w9WgXcQ"
})
data = res.json()
print(data["title"], data["thumbnail"])
const res = await fetch("{{ request.host_url.rstrip('/') }}/api/analyze", {
  method: "POST",
  headers: { "Content-Type": "application/json" },

});
const data = await res.json();

GET /api/status/<download_id> Poll download progress — keep calling until status is completed 
curl {{ request.host_url.rstrip('/') }}/api/status/<download_id>
import requests, time

download_id = "a1b2c3d4-..."
while True:
    res = requests.get(f"{{ request.host_url.rstrip('/') }}/api/status/{download_id}")
    data = res.json()
    print(data["status"], data.get("progress", ""))
    if data["status"] in ("completed", "error"):
        break
    time.sleep(2)
async function pollStatus(id) {
  while (true) {
    const res = await fetch(`{{ request.host_url.rstrip('/') }}/api/status/${id}`);
    const data = await res.json();
    console.log(data.status, data.progress);
    if (data.status === "completed" || data.status === "error") break;
    await new Promise(r => setTimeout(r, 2000));
  }
}

Full Flow Example (Python)
import requests, time

BASE = "{{ request.host_url.rstrip('/') }}"
VIDEO_URL = "https://www.youtube.com/watch?v=dQw4w9WgXcQ"

# Step 1: Analyze
info = requests.post(f"{BASE}/api/analyze", json={"url": VIDEO_URL}).json()
print(f"Title: {info['title']} | Platform: {info['platform']}")

# Step 2: Start download
dl = requests.post(f"{BASE}/api/download", json={"url": VIDEO_URL, "format": "best"}).json()
download_id = dl["id"]

# Step 3: Poll status
while True:
    status = requests.get(f"{BASE}/api/status/{download_id}").json()
    print(f"Status: {status['status']} | Progress: {status.get('progress', 0)}%")
    if status["status"] == "completed":
        filename = status["filename"]
        break
    elif status["status"] == "error":
        raise Exception(status.get("error", "Download failed"))
    time.sleep(2)

# Step 4: Save file
res = requests.get(f"{BASE}/file/{download_id}", stream=True)
with open(filename, "wb") as f:
    for chunk in res.iter_content(8192):
        f.write(chunk)
print(f"Saved: {filename}")
